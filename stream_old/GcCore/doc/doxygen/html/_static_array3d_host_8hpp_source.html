103  template<typename T, uint32_t sizeX, uint32_t sizeY, uint32_t sizeZ, uint32_t Option>
104  StaticArray3dHost<T, sizeX, sizeY, sizeZ, Option>::StaticArray3dHost()
105  {
106  _data = nullptr;
107  switch (Option)
108  {
109  case StaticArrayOptions::Options::Pinned:
110  // Allocation of the array on the CPU with CUDA using pinned memory
111  CUDA_SAFE_CALL(cudaMallocHost(
112  reinterpret_cast<void**> (&_data),
113  sizeX * sizeY * sizeZ * sizeof(T)));
114  break;
115 
116  case StaticArrayOptions::Options::Mapped:
117  // Allocation of the array on the CPU with CUDA using GPU mapped memory
118  CUDA_SAFE_CALL(cudaHostAlloc(
119  reinterpret_cast<void**> (&_data),
120  sizeX * sizeY * sizeZ * sizeof(T),
121  cudaHostAllocMapped | cudaHostAllocWriteCombined));
122  break;
123 
124  case StaticArrayOptions::Options::Standard:
125  // Allocation of the array on the CPU in a classic way
126  _data = new T[sizeX * sizeY * sizeZ];
127  break;
128  }
129  }
