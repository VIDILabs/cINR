366  {
367  uint3 brickPosition;
368  tdns::common::K_DynamicArray3dDevice<uint3> levelDimensions;
369  tdns::common::K_DynamicArray3dDevice<uint32_t> levelCoordinates;
370 
371  uint32_t coordinates;
372  uint3 dimensions;
373 
374  // if we get multiple level in the page table hierarchy
375  if (_nbTableCaches != 0)
376  {
377  levelDimensions = _tableCaches[_nbTableCaches - 1].get_real_number_of_entries();
378  levelCoordinates = _tableCaches[_nbTableCaches - 1].get_level_coordinates();
379  }
380  else // only the MRPD
381  {
382  levelDimensions = _mrpd.get_level_dimensions();
383  levelCoordinates = _mrpd.get_level_coordinates();
384  }
385 
386  // We just need the x dimensions because all the levels are summed on x and y, z are the size max
387  dimensions = levelDimensions(make_uint3(level, 0, 0));
388  // offset on x, according to the level
389  coordinates = levelCoordinates(make_uint3(level, 0, 0));
390 
391  brickPosition.x = coordinates + static_cast<uint32_t>(position.x * dimensions.x);
392  brickPosition.y = static_cast<uint32_t>(position.y * dimensions.y);
393  brickPosition.z = static_cast<uint32_t>(position.z * dimensions.z);
394 
395  _requestBuffer(brickPosition) = _timeStamp;
