102  inline DynamicArray3dHost<T, Option>::DynamicArray3dHost(const tdns::math::Vector3ui &size)
103  {
104  _data = nullptr;
105  _size = size;
106  _xyProduct = size[0] * size[1];
107 
108  switch (Option)
109  {
110  case DynamicArrayOptions::Options::Pinned:
111  // Allocation of the array on the CPU with CUDA using pinned memory
112  CUDA_SAFE_CALL(cudaMallocHost(
113  reinterpret_cast<void**> (&_data),
114  size[0] * size[1] * size[2] * sizeof(T)));
115  break;
116 
117  case DynamicArrayOptions::Options::Mapped:
118  // Allocation of the array on the CPU with CUDA using GPU mapped memory
119  CUDA_SAFE_CALL(cudaHostAlloc(
120  reinterpret_cast<void**> (&_data),
121  size[0] * size[1] * size[2] * sizeof(T),
122  cudaHostAllocMapped | cudaHostAllocWriteCombined));
123  break;
124 
125  case DynamicArrayOptions::Options::Standard:
126  // Allocation of the array on the CPU in a classic way
127  _data = new T[size[0] * size[1] * size[2]];
128  break;
129  }
130  }
