204  thrust::device_vector<LruContent> result(_lru.size());
205 
206  // stream compaction to keep the used elements
207  thrust::device_vector<LruContent>::iterator it = thrust::copy_if(
208  _lru.begin(),

210  masks.begin(),
211  result.begin(),
212  predicate::equal<U>(ref));
213 
214  // stream compaction to keep the unused elements
215  thrust::copy_if(_lru.begin(), _lru.end(), masks.begin(), it, predicate::not_equal<U>(ref));
216 
217  // update the LRU with the most recently used elements at the begining
218  _lru = result;
