316  uint3 K_CacheDevice<T>::compute_element_position(const uint4 &tableEntry, uint32_t level, const float3 &position, uint32_t covering)
317  {
318  uint3 virtualizedNumberOfEntries = _virtualizedNumberOfEntries[level];
319  uint3 blockSize = _blockSize;
320 
321  uint3 elementPosition = make_uint3(tableEntry.x, tableEntry.y, tableEntry.z);
322  // elementPosition = position in the surface of the begining of the element
323 
324  elementPosition.x /= blockSize.x;
325  elementPosition.y /= blockSize.y;
326  elementPosition.z /= blockSize.z;
327 
328  // elementPosition = position in the "usage buffer"
329  update_usage_buffer(elementPosition);
330 
331  elementPosition.x = covering + tableEntry.x + static_cast<uint32_t>((position.x * virtualizedNumberOfEntries.x)) % (blockSize.x - 2 * covering);
332  elementPosition.y = covering + tableEntry.y + static_cast<uint32_t>((position.y * virtualizedNumberOfEntries.y)) % (blockSize.y - 2 * covering);
333  elementPosition.z = covering + tableEntry.z + static_cast<uint32_t>((position.z * virtualizedNumberOfEntries.z)) % (blockSize.z - 2 * covering);
334
