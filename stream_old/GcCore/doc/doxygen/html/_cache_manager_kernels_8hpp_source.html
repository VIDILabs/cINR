222  LruContent *stack,

235  uint4 entry;
236 
237  while (true) // going up to update the hierarchy
238  {
239  if (i < 0 || stack[i].volumePosition.x < -1.f) break; // end of loop
240 
241  if (i == 0)//==> mrpd
242  {
243  entry = mrpd.get(level, position);
244  if (entry.w >= 1) break; //end of update - already mapped or empty
245  const uint3 &cacheEntry = stack[i + 1].cachePosition;
246  mrpd.set(level, position, make_uint4(
247  cacheEntry.x,
248  cacheEntry.y,
249  cacheEntry.z,
250  empty ? 2 : 1));
251  }
252  else
253  {
254  //check if we need to add a new entry or complete an existing one
255  if (stack[i - 1].volumePosition.x < -1.f) //complete an existing one
256  {
257  const uint3 &cacheEntry = stack[i - 1].cachePosition;
258  //flag the new entry
259  tableCaches[i - 1].insert(
260  make_uint4(cacheEntry.x, cacheEntry.y, cacheEntry.z, 0),
261  level,
262  position,

264  break;
265  }
266

268  const uint3 &cacheEntry = stack[i].cachePosition;
269 
270  // reset entry in the current cache
271  tableCaches[i - 1].reset_entries(cacheEntry);
272 
273  if (stack[i].volumePosition.x > -1.f)
274  {
275  //unflag from the previous cache. We must go back from the top level (the MRPD)
276  uint4 tmpEntry = mrpd.get(stack[i].level, stack[i].volumePosition);
277  if (i == 1)
278  mrpd.set(stack[i].level, stack[i].volumePosition, make_uint4(0, 0, 0, 0));
279  else
280  {
281  for (uint32_t j = 0; j < i - 1; ++j)
282  tmpEntry = tableCaches[j].get(tmpEntry, stack[i].level, stack[i].volumePosition);
283 
284  tableCaches[i - 2].insert(
285  tmpEntry,
286  stack[i].level,
287  stack[i].volumePosition,
288  make_uint4(0, 0, 0, 0));
289  }
290  }
291 
292  //flag the new entry
293  tableCaches[i - 1].insert(
294  make_uint4(cacheEntry.x, cacheEntry.y, cacheEntry.z, 0),
295  level,
296  position,
297  make_uint4(stack[i + 1].cachePosition.x, stack[i + 1].cachePosition.y, stack[i + 1].cachePosition.z, 1));
298 
299  // LRU update
300  lrus[i - 1][(lruSizes[i - 1] - 1) - indexLRU[i - 1] + 1].volumePosition = position;
301  lrus[i - 1][(lruSizes[i - 1] - 1) - indexLRU[i - 1] + 1].level = level;
302  }

304  }
